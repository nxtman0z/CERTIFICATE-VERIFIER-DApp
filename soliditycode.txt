pragma solidity ^0.8.0;

contract CertificateVerifier {
    struct Certificate {
        string ipfsHash;
        uint256 issuedAt;
    }

    mapping(address => Certificate[]) public certificates;

    event CertificateIssued(address indexed to, string ipfsHash);

    function issueCertificate(address _to, string memory _ipfsHash) public {
        certificates[_to].push(Certificate(_ipfsHash, block.timestamp));
        emit CertificateIssued(_to, _ipfsHash);
    }

    function getCertificates(address _owner) public view returns (Certificate[] memory) {
        return certificates[_owner];
    }
}
